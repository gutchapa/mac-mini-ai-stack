    <script>
        let feeEntries = [];
        let entryCount = 0;

        // --- DOM Elements ---
        const addEntryBtn = document.getElementById('addEntryBtn');
        const generateReceiptBtn = document.getElementById('generateReceiptBtn');
        const printBtn = document.getElementById('printBtn');
        const feeTableBody = document.getElementById('feeTableBody');

        // --- Event Listeners ---

        addEntryBtn.addEventListener('click', addFeeEntry);
        generateReceiptBtn.addEventListener('click', generateReceipt);
        printBtn.addEventListener('click', () => window.print());

        // --- Functions ---

        function addFeeEntry() {
            const name = document.getElementById('studentName').value;
            const section = document.getElementById('classSection').value;
            const date = document.getElementById('date').value;
            const description = document.getElementById('feeDescription').value;
            const amount = parseFloat(document.getElementById('amount').value);

            if (!name || !section || !date || !description || isNaN(amount) || amount <= 0) {
                alert("Please fill in all fields correctly.");
                return;
            }

            const newEntry = {
                id: entryCount + 1,
                name: name,
                section: section,
                date: date,
                description: description,
                amount: amount
            };

            feeEntries.push(newEntry);
            renderTable();
            
            // Clear form fields (optional, but good UX)
            document.getElementById('amount').value = '';
            document.getElementById('feeDescription').value = '';
            entryCount++;
        }

        function renderTable() {
            feeTableBody.innerHTML = ''; // Clear existing rows

            if (feeEntries.length === 0) {
                feeTableBody.innerHTML = '<tr><td colspan="3" style="text-align: center;">No entries added yet.</td></tr>';
                return;
            }

            feeEntries.forEach(entry => {
                const row = feeTableBody.insertRow();
                row.insertCell(0).textContent = entry.id;
                row.insertCell(1).textContent = entry.description;
                row.insertCell(2).textContent = entry.amount.toFixed(2);
            });
        }

        function generateReceipt() {
            if (feeEntries.length === 0) {
                alert("Please add fee entries before generating a receipt.");
                return;
            }

            // 1. Calculate Total
            const totalAmount = feeEntries.reduce((sum, entry) => sum + entry.amount, 0);

            // 2. Generate Receipt HTML
            let receiptHTML = `
                <div style="border: 1px solid #ccc; padding: 20px; font-family: Arial, sans-serif;">
                    <div style="text-align: center; margin-bottom: 20px;">
                        <h2 style="color: var(--navy);">Gutchapa Academy</h2>
                        <div style="border-bottom: 2px solid var(--gold); margin: 5px 0;"></div>
                    </div>

                    <div class="receipt-detail">
                        <strong>Student Name:</strong> ${feeEntries[0].name} <br>
                        <strong>Class/Section:</strong> ${feeEntries[0].section} <br>
                        <strong>Issue Date:</strong> ${new Date().toLocaleDateString()} <br>
                        <strong>Receipt No:</strong> REC-${String(Math.floor(Math.random() * 9000) + 1000).toString().padStart(4, '0')}}
                    </div>

                    <h3 style="margin-top: 20px; border-bottom: 1px solid #eee;">Fee Details</h3>
            `;

            // 3. List Entries
            feeEntries.forEach(entry => {
                receiptHTML += `
                    <div class="receipt-detail">
                        <strong>${entry.description}:</strong> ${entry.amount.toFixed(2)} INR
                    </div>
                `;
            });

            // 4. Total
            receiptHTML += `
                    <div style="margin-top: 20px; padding-top: 10px; border-top: 2px dashed var(--navy);">
                        <h3 style="text-align: right;">TOTAL AMOUNT: ${totalAmount.toFixed(2)} INR</h3>
                    </div>
                </div>
            `;

            // 5. Display Receipt (using a temporary window for printing)
            const receiptWindow = window.open('', '_blank');
            receiptWindow.document.write(`
                <html>
                <head>
                    <title>Fee Receipt</title>
                    <style>
                        body { font-family: Arial, sans-serif; margin: 30px; }
                        h2 { color: navy; }
                        .receipt-header { border-bottom: 2px solid gold; padding-bottom: 5px; margin-bottom: 15px; }
                        .receipt-detail { margin-bottom: 10px; }
                    </style>
                </head>
                <body>
                    ${receiptHTML}
                </body>
                </html>
            `);
            receiptWindow.document.close();
            receiptWindow.focus();
        }

        // Initialize the table on load
        renderTable();
    </script>

</body>
</html>
