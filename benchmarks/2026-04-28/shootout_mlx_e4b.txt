    <script>
        // --- DOM Elements ---
        const feeBody = document.getElementById('feeBody');
        const addBtn = document.getElementById('addBtn');
        const generateBtn = document.getElementById('generateBtn');
        const printBtn = document.getElementById('printBtn');
        const feeDescSelect = document.getElementById('feeDesc');
        const amountInput = document.getElementById('amount');
        const studentNameInput = document.getElementById('studentName');
        const classSectionInput = document.getElementById('classSection');
        const currentTotalSpan = document.getElementById('currentTotal');
        const receiptViewDiv = document.getElementById('receiptView');

        // --- State Variables ---
        let feeEntries = [];
        let receiptNumber = '';

        // --- Utility Functions ---

        /** Formats a number to Indian currency style (2 decimal places) */
        const formatCurrency = (amount) => `₹ ${parseFloat(amount).toFixed(2)}`;

        /** Generates a unique receipt number (e.g., REC-0001) */
        const generateReceiptId = () => {
            const prefix = 'REC-';
            // Simple sequential ID generation for this client-side demo
            const count = feeEntries.length + 1;
            const paddedId = String(count).padStart(4, '0');
            return prefix + paddedId;
        };

        /** Updates the displayed total and the table body */
        const updateDisplay = () => {
            feeBody.innerHTML = '';
            let total = 0;
            
            feeEntries.forEach((entry, index) => {
                const row = `
                    <tr>
                        <td>${index + 1}</td>
                        <td>${entry.description}</td>
                        <td>${formatCurrency(entry.amount)}</td>
                    </tr>
                `;
                feeBody.insertAdjacentHTML('beforeend', row);
                total += parseFloat(entry.amount);
            });

            currentTotalSpan.textContent = formatCurrency(total);
        };

        // --- Event Handlers ---

        /** Handles adding a new fee item from the form */
        const handleAddItem = () => {
            const name = studentNameInput.value.trim();
            const section = classSectionInput.value.trim();
            const description = feeDescSelect.value;
            const amountStr = amountInput.value.trim();

            if (!name || !section || !description || !amountStr || parseFloat(amountStr) <= 0) {
                alert("Please fill in all fields correctly and ensure the amount is greater than zero.");
                return;
            }

            const amount = parseFloat(amountStr);

            feeEntries.push({
                description: description,
                amount: amount
            });

            updateDisplay();

            // Clear form fields for next entry
            feeDescSelect.value = "";
            amountInput.value = "";
            // Keep name/class filled if adding multiple items for the same student
            // For simplicity, we clear all inputs after a successful add.
            studentNameInput.value = "";
            classSectionInput.value = "";
            amountInput.focus();
        };

        /** Generates the printable receipt view */
        const handleGenerateReceipt = () => {
            if (feeEntries.length === 0) {
                alert("Please add fee items before generating the receipt.");
                return;
            }

            const totalAmount = feeEntries.reduce((sum, entry) => sum + parseFloat(entry.amount), 0);
            const currentDate = new Date().toLocaleDateString('en-US');
            
            // Use the first entry's details for the main header details
            const studentName = feeEntries[0].name || studentNameInput.value.trim();
            const classSection = feeEntries[0].class || classSectionInput.value.trim();

            receiptNumber = generateReceiptId();

            // Build the detailed table HTML for the receipt
            const receiptBodyHtml = feeEntries.map((entry, index) => `
                <tr>
                    <td>${index + 1}</td>
                    <td>${entry.description}</td>
                    <td>${formatCurrency(entry.amount)}</td>
                </tr>
            `).join('');

            // Construct the full receipt HTML
            const receiptHtml = `
                <div class="receipt-header">
                    <h1 class="school-name">Gutchapa Academy</h1>
                    <div class="gold-accent"></div>
                    <div class="receipt-details">
                        <p><strong>RECEIPT NO:</strong> ${receiptNumber}</p>
                        <p><strong>ISSUE DATE:</strong> ${currentDate}</p>
                        <hr style="border: none; border-top: 1px solid #ccc; margin: 15px 0;">
                        <p><strong>Student Name:</strong> ${studentName}</p>
                        <p><strong>Class/Section:</strong> ${classSection}</p>
                    </div>
                </div>

                <div class="receipt-body">
                    <table>
                        <thead>
                            <tr>
                                <th>#</th>
                                <th>Description</th>
                                <th>Amount</th>
                            </tr>
                        </thead>
                        <tbody>
                            ${receiptBodyHtml}
                        </tbody>
                    </table>
                    <div class="receipt-footer">
                        TOTAL PAID: ${formatCurrency(totalAmount)}
                    </div>
                </div>
            `;

            // Display the receipt
            receiptViewDiv.innerHTML = receiptHtml;
            receiptViewDiv.style.display = 'block';
            
            // Switch views and enable print button
            document.body.scrollTo(0, 0); // Scroll to top
            document.getElementById('feeListTable').style.display = 'none';
            document.querySelector('.input-section:nth-of-type(2)').style.display = 'none';
            document.querySelector('.input-section:nth-of-type(1)').style.display = 'none';
            
            document.getElementById('receiptView').style.display = 'block';
            printBtn.style.display = 'inline-block';
            
            // Scroll to the receipt view
            receiptViewDiv.scrollIntoView({ behavior: 'smooth' });
        };

        // --- Event Listeners ---
        addBtn.addEventListener('click', handleAddItem);
        generateBtn.addEventListener('click', handleGenerateReceipt);
        
        // Print button triggers window.print() on the receipt view
        printBtn.addEventListener('click', () => {
            window.print();
        });

        // Allow pressing Enter in amount field to trigger Add
        amountInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                handleAddItem();
            }
        });

    </script>
